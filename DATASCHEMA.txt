USER TABLE:
CREATE TABLE USER(
UID NUMBER(4) PRIMARY KEY , 
PHONE NUMBER(10) NOT NULL UNIQUE ,
EMAIL EMAILID NOT NULL UNIQUE, 
NAME VARCHAR(40) NOT NULL,  
PASSWORD PASSWORDFIELD NOT NULL,  
ADDRESS VARCHAR(50) NOT NULL,
CHECK(LENGTH(PASSWORD)>8), 
CHECK(NAME DOES NOT CONTAIN [0-9] OR ANY SPECIAL CHARACTHER)
);

PAYMENT TABLE:
CREATE TABLE PAYMENT(
PID NUMBER(4) PRIMARY KEY, AMOUNT NUMBBER(10) NOT NULL,  
FOREIGN KEY(UID) REFERENCES USER(UID) ON UPDATE CASCADE, 
DAT DATE NOT NULL, 
FOREIGN KEY(RID) REFERENCES RENT(RID) ON UPDATE CASCADE
);

RENT TABLE:
CREATE TABLE RENT(
    RID NUMBER(4) PRIMARY KEY, TOTAL NUMBER(10) NOT NULL, 
    GIVEN NUMBER(10) DEFAULT 0, 
    NOD NUMBER(4) NOT NULL DEFAULT 1, START DATE NOT NULL, 
    FOREIGN KEY(UID) REFERENCES USER(UID) ON UPDATE CASCADE,
    FOREIGN KEY(VNO) REFERENCES CARS(VNO) ON UPDATE CASCADE
);

CAR TABLE:
CREATE TABLE CARS(
    VNO NUMBER(4) PRIMARY KEY, MODEL VARCHAR2(20) NOT NULL UNIQUE,
    RATE NUMBER(5) NOT NULL,  NOS NUMBER(2) DEFAULT 4, 
    OCCUP BOOLEAN NOT NULL DEFAULT FALSE
);
